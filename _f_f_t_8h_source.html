105  typedef T fftw_type;

116  // Handle the FFTW3 memory allocation, which assured 16-bit alignment for SSE usage.
117  // FFTW3 now states that C++ complex<double> will be bit-compatible with
118  // the fftw_complex type. So all interfaces will be through std::complex<double>
119  // And the fftw real type is now just double.
120  template <typename T>
121  class FFTW_Array
122  {
123  typedef typename FFTW_Traits<T>::fftw_type fftw_type;
124  public:
125  FFTW_Array() : _n(0), _p(0) {}
126 
127  FFTW_Array(size_t n) : _n(0), _p(0)
128  { resize(n); }
129 
130  FFTW_Array(size_t n, T val) : _n(0), _p(0)
131  {
132  resize(n);
133  fill(val);
134  }
135 
136  FFTW_Array(const FFTW_Array<T>& rhs) : _n(0), _p(0)
137  {
138  resize(rhs._n);
139  for (size_t i=0; i<_n; ++i) _p[i] = rhs._p[i];
140  }
141 
142  FFTW_Array<T>& operator=(const FFTW_Array<T>& rhs)
143  {
144  if (this != &rhs) {
145  resize(rhs._n);
146  for (size_t i=0; i<_n; ++i) _p[i] = rhs._p[i];
147  }
148  return *this;
149  }
150 
151  ~FFTW_Array();
152 
153  void resize(size_t n);
154 
155  void fill(T val)
156  {
157  for (size_t i=0; i<_n; ++i) _p[i] = val;
158  }
159 
160  size_t size() { return _n; }
161 
162  T* get() { return _p; }
163  const T* get() const { return _p; }
164 
165  fftw_type* get_fftw()
166  { return reinterpret_cast<fftw_type*>(_p); }
167  const fftw_type* get_fftw() const
168  { return reinterpret_cast<const fftw_type*>(_p); }
169 
170  T& operator[](size_t i) { return _p[i]; }
171  const T& operator[](size_t i) const { return _p[i]; }
172 
173  private:
174  size_t _n;
175  T* _p;
176  };

502  template <class T>
503  void KTable::fill(const T& f)
504  {
505  clearCache(); // invalidate any stored interpolations
506  std::complex<double>* zptr=_array.get();
507  double kx, ky;
508  for (int iy=0; iy< _No2; iy++) {
509  ky = iy*_dk;
510  for (int ix=0; ix< _No2+1 ; ix++) {
511  kx = ix*_dk;
512  *(zptr++) = f(kx,ky);
513  }
514  }
515  // wrap to the negative ky's
516  for (int iy=-_No2; iy< 0; iy++) {
517  ky = iy*_dk;
518  for (int ix=0; ix< _No2+1 ; ix++) {
519  kx = ix*_dk;
520  *(zptr++) = f(kx,ky);
521  }
522  }
523  return;
524  }
