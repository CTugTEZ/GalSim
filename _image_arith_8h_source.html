45  template <typename T, typename Op>
46  Op for_each_pixel(const BaseImage<T>& image, Op f)
47  {
48  const T* ptr = image.getData();
49  if (ptr) {
50  const int skip = image.getNSkip();
51  const int step = image.getStep();
52  const int nrow = image.getNRow();
53  const int ncol = image.getNCol();
54  if (step == 1) {
55  for (int j=0; j<nrow; j++, ptr+=skip)
56  for (int i=0; i<ncol; i++) f(*ptr++);
57  } else {
58  for (int j=0; j<nrow; j++, ptr+=skip)
59  for (int i=0; i<ncol; i++, ptr+=step) f(*ptr);
60  }
61  }
62  return f;
63  }

116  template <typename T1, typename T2, typename Op>
117  Op transform_pixel(ImageView<T1> image1, const BaseImage<T2>& image2, Op f)
118  {
119  T1* ptr1 = image1.getData();
120  if (ptr1) {
121 
122  if (!image1.getBounds().isSameShapeAs(image2.getBounds()))
123  throw ImageError("transform_pixel image bounds are not same shape");
124 
125  const int skip1 = image1.getNSkip();
126  const int step1 = image1.getStep();
127  const int nrow = image1.getNRow();
128  const int ncol = image1.getNCol();
129  const T2* ptr2 = image2.getData();
130  const int skip2 = image2.getNSkip();
131  const int step2 = image2.getStep();
132  if (step1 == 1 && step2 == 1) {
133  for (int j=0; j<nrow; j++, ptr1+=skip1, ptr2+=skip2)
134  for (int i=0; i<ncol; i++, ++ptr1, ++ptr2) *ptr1 = f(*ptr1,T1(*ptr2));
135  } else {
136  for (int j=0; j<nrow; j++, ptr1+=skip1, ptr2+=skip2)
137  for (int i=0; i<ncol; i++, ptr1+=step1, ptr2+=step2) *ptr1 = f(*ptr1,T1(*ptr2));
138  }
139  }
140  return f;
141  }
