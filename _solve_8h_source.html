188  void bracket()
189  {
190  const T factor=2.0;
191  if (uBound == lBound)
192  throw SolveError("uBound=lBound in bracket()");
193  evaluateBounds();
194 
195  T delta = uBound-lBound;
196  for (int j=1; j<maxSteps; j++) {
197  if (fupper*flower < 0.0) return;
198  if (std::abs(flower) < std::abs(fupper)) {
199  uBound = lBound;
200  fupper = flower;
201  delta *= factor;
202  lBound -= delta;
203  flower = func(lBound);
204  } else {
205  lBound = uBound;
206  flower = fupper;
207  delta *= factor;
208  uBound += delta;
209  fupper = func(uBound);
210  }
211  }
212  throw SolveError("Too many iterations in bracket()");
213  }

261  // General purpose one-sided bracket with limit.
262  // Used by bracketUpperWithLimit and bracketLowerWithLimit.
263  // Expand in the direction of b.
264  bool bracket1WithLimit(T& a, T& b, T& fa, T& fb, T& c)
265  {
266  const T factor=2.0;
267  // The principal here is to use z = (b-a)/(b-c) as our variable to double each time
268  // where a is the initial lBound, c is the upper_limit, and b is the uBound we
269  // are trying to find.
270  //
271  // z = (b-a)/(b-c)
272  // b-a = zb-zc
273  // b = (zc-a)/(z-1)
274  //
275  // So if we take z' = f*z
276  // z' = f(b-a)/(b-c)
277  // Then
278  // b' = (z'c-a)/(z'-1)
279  // b' = (fcb-fca-ab+ac)/(b-c) / (fb-fa-b+c)/(b-c)
280  // = ((fc-a)b-(f-1)ac) / ((f-1)b-fa+c)
281  // = (fc(b-a) + a(c-b)) / (f(b-a) + (c-b))
282  // This is our fundamental iteration formula.
283  //
284  // To see the behavior of this iteration, take two limiting cases.
285  // 1) If b = a+d and d << c-a, and use f=2 for simplicity:
286  // b' = (2cd + ac - a^2 - ad)) / (2d+c-a-d)
287  // = (a(c-a) + d(2c-a)) / (c-a+d)
288  // = (a + d(2c-a)/(c-a)) / (1+d/(c-a))
289  // = (a + d(2c-a)/(c-a)) * (1-d/(c-a))
290  // = a + d(2c-a)/(c-a) - da/(c-a)
291  // = a + d(2c-2a)/(c-a)
292  // = a + 2d
293  //
294  // 2) If b = c-d and d << c-a, and use f=2 for simplicity:
295  // b' = (2c^2-2cd-2ca+ad) / (2c-2d-2a+d)
296  // = (2c(c-a) - d(2c-a)) / (2(c-a)-d)
297  // = (c - d(2c-a)/(2c-2a)) / (1-d/(2c-2a))
298  // = (c - d(2c-a)/(2c-2a)) * (1+d/(2c-2a))
299  // = c - d(2c-a)/2(c-a) + cd/2(c-a)
300  // = c - d(c-a)/2(c-a)
301  // = c - d/2
302  //
303  // So when far from the limit (1), the iteration is just like the version without
304  // the limit. But when the variable approaches the limit (2), it just goes half the
305  // remaining distance each time.
306 
307  xdbg<<"Bracket with limits:\n";
308  xdbg<<"a,b,c = "<<a<<','<<b<<','<<c<<std::endl;
309  xdbg<<"fa,fb = "<<fa<<','<<fb<<std::endl;
310  for (int j=1; j<maxSteps; j++) {
311  if (fa*fb < 0.0) return true;
312  T bma = b-a; // Do this before overwriting a!
313  T cmb = c-b;
314  a = b;
315  fa = fb;
316  // We divide the top and bottom by (b-a)(c-b) for stability in case one of these
317  // is especially larger than the other.
318  b = (factor*c/cmb + a/bma) / (factor/cmb + 1./bma);
319  xdbg<<"b -> "<<b<<std::endl;
320  fb = func(b);
321  xdbg<<"fb -> "<<fb<<std::endl;
322  }
323  return false;
324  }

376  T bisect() const
377  {
378  T dx,f,fmid,xmid,rtb;
379 
380  evaluateBounds();
381  f=flower;
382  fmid=fupper;
383 
384  if (f*fmid >= 0.0)
385  FormatAndThrow<SolveError> () << "Root is not bracketed: " << lBound
386  << " " << uBound;
387  rtb = f < 0.0 ? (dx=uBound-lBound,lBound) : (dx=lBound-uBound,uBound);
388  for (int j=1;j<=maxSteps;j++) {
389  fmid=func(xmid=rtb+(dx *= 0.5));
390  if (fmid <= 0.0) rtb=xmid;
391  if ( (std::abs(dx) < xTolerance) || fmid == 0.0) return rtb;
392  }
393  throw SolveError("Too many bisections");
394  return 0.0;
395  }

400  T zbrent() const
401  {
402  T a=lBound, b=uBound, c=uBound;
403  T d=b-a, e=b-a;
404  T min1,min2;
405  evaluateBounds();
406  T fa = flower;
407  T fb = fupper;
408 
409  T p,q,r,s,tol1,xm;
410  if ((fa > 0.0 && fb > 0.0) || (fa < 0.0 && fb < 0.0)) {
411  FormatAndThrow<SolveError> () << "Root is not bracketed: "
412  << lBound << " " << uBound;
413  }
414  T fc=fb;
415  for (int iter=0;iter<=maxSteps;iter++) {
416  if ((fb > 0.0 && fc > 0.0) || (fb < 0.0 && fc < 0.0)) {
417  c=a;
418  fc=fa;
419  e=d=b-a;
420  }
421  if (std::abs(fc) < std::abs(fb)) {
422  a=b;
423  b=c;
424  c=a;
425  fa=fb;
426  fb=fc;
427  fc=fa;
428  }
429  tol1=2.0*std::numeric_limits<T>::epsilon()*std::abs(b) + 0.5*xTolerance;
430  xm=0.5*(c-b);
431  if (std::abs(xm) <= tol1 || fb == 0.0) return b;
432  if (std::abs(e) >= tol1 && std::abs(fa) > std::abs(fb)) {
433  s=fb/fa;
434  if (a == c) {
435  p=2.0*xm*s;
436  q=1.0-s;
437  } else {
438  q=fa/fc;
439  r=fb/fc;
440  p=s*(2.0*xm*q*(q-r)-(b-a)*(r-1.0));
441  q=(q-1.0)*(r-1.0)*(s-1.0);
442  }
443  if (p > 0.0) q = -q;
444  p=std::abs(p);
445  min1=3.0*xm*q-std::abs(tol1*q);
446  min2=std::abs(e*q);
447  if (2.0*p < std::min(min1,min2) ) {
448  e=d;
449  d=p/q;
450  } else {
451  d=xm;
452  e=d;
453  }
454  } else {
455  d=xm;
456  e=d;
457  }
458  a=b;
459  fa=fb;
460  if (std::abs(d) > tol1) b += d;
461  else b += (xm>=0. ? std::abs(tol1) : -std::abs(tol1));
462  fb=func(b);
463  }
464  throw SolveError("Maximum number of iterations exceeded in zbrent");
465  }
