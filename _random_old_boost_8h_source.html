40  // Uniform wrapper is given arguments to map RNG's to [0.,1.) interval.

42  public:
43  UniformDeviate(): urd(0.,1.) { seedtime(); } // seed with time
44  UniformDeviate(const long lseed): urng(lseed), urd(0.,1.) {} //seed with specific number
45  double operator() () { return urd(urng); }
46  operator double() { return urd(urng); }

48  void seed(const long lseed) { urng.seed(lseed); }

50  private:
51  OurURNG urng;
52  boost::uniform_real<> urd;
53  void seedtime()
54  {
55  struct timeval tp;
56  gettimeofday(&tp,NULL);
57  urng.seed(tp.tv_usec);
58  }

70  // A Gaussian deviate (unit variance, zero-mean by default).
71  // Wraps the Boost.Random normal_distribution so that
72  // the parent UniformDeviate is given once at construction, and copy/assignment are hidden.
73  class GaussianDeviate
74  {
75  typedef boost::variate_generator<OurURNG&,
76  boost::normal_distribution<> > GaussGenerator;
77  public:
78  GaussianDeviate(UniformDeviate& u_, double mean, double sigma) :
79  u(u_),
80  normal(new boost::normal_distribution<>(mean,sigma)),
81  gen(new GaussGenerator(u.urng,*normal)) {}
82  ~GaussianDeviate() {if (gen) delete gen; if (normal) delete normal;}
83  double operator() () { return (*gen)(); }
84  operator double() { return (*gen)();}
85  double getMean() {return normal->mean();}
86  double getSigma() {return normal->sigma();}
87  void setMean(double mean) {
88  boost::normal_distribution<>* normal2 =
89  new boost::normal_distribution<>(mean,normal->sigma());
90  delete normal;
91  delete gen;
92  normal = normal2;
93  gen = new GaussGenerator(u.urng,*normal);
94  }
95  void setSigma(double sigma) {
96  boost::normal_distribution<>* normal2 =
97  new boost::normal_distribution<>(normal->mean(),sigma);
98  delete normal;
99  delete gen;
100  normal = normal2;
101  gen = new GaussGenerator(u.urng,*normal);
102  }
103  private:
104  UniformDeviate& u;
105  boost::normal_distribution<>* normal;
106  GaussGenerator* gen;
107  // Hide:
108  GaussianDeviate(const GaussianDeviate& rhs): u(rhs.u), gen(0), normal(0) {}
109  void operator=(const GaussianDeviate& rhs) {}
110  };
