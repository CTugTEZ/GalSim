67  // Function for applying deviates to an image... Used as a method for all Deviates below.
68  template <typename D, typename T>
69  static void ApplyDeviateToImage(D& dev, ImageView<T>& data)
70  {
71  // Typedef for image row iterable
72  typedef typename ImageView<T>::iterator ImIter;
73 
74  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
75  ImIter ee = data.rowEnd(y);
76  for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it += T(dev()); }
77  }
78  }
