40  // Function for applying deviates to an image... Used as a method for all Deviates below.
41  template <typename D, typename T>
42  static void ApplyDeviateToImage(D& dev, ImageView<T>& data)
43  {
44  // Typedef for image row iterable
45  typedef typename ImageView<T>::iterator ImIter;
46 
47  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
48  ImIter ee = data.rowEnd(y);
49  for (ImIter it = data.rowBegin(y); it != ee; ++it) { *it += T(dev()); }
50  }
51  }
