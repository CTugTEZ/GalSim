480  template <typename T>
481  void applyToView(ImageView<T> data)
482  {
483  data += T(_sky_level);
484 
485  // Add the Poisson noise first:
486  if (_gain > 0.) {
487  PoissonDeviate pd(*_rng, 1.); // will reset the mean for each pixel below.
488  SkyNoiseAdder<T> adder(pd, _gain);
489  transform_pixel(data, adder);
490  }
491 
492  // Next add the Gaussian noise:
493  if (_read_noise > 0.) {
494  GaussianDeviate gd(*_rng, 0., _read_noise / (_gain > 0. ? _gain : 1.));
495  ReadNoiseAdder<T> adder(gd);
496  transform_pixel(data, adder);
497  }
498 
499  data -= T(_sky_level);
500  }
