283  template <typename T>
284  void applyToView(ImageView<T> data)
285  {
286  // Above this many e's, assume Poisson distribution == Gaussian
287  // The Gaussian deviate is about 20% faster than Poisson, and for high N
288  // they are virtually identical.
289  const double MAX_POISSON=1.e5;
290  // Typedef for image row iterable
291  typedef typename ImageView<T>::iterator ImIter;
292 
293  data += T(_sky_level);
294 
295  PoissonDeviate pd(*_rng, 1.); // will reset the mean for each pixel below.
296  GaussianDeviate gd(*_rng, 0., 1.);
297  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
298  ImIter ee = data.rowEnd(y);
299  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
300  if (*it <= 0.) continue;
301  if (*it < MAX_POISSON) {
302  pd.setMean(*it);
303  *it = T(pd());
304  } else {
305  gd.setSigma(sqrt(*it));
306  *it = T(*it + gd());
307  }
308  }
309  }
310 
311  data -= T(_sky_level);
312  }

449  void applyToView(ImageView<T> data)
450  {
451  // Above this many e's, assume Poisson distribution == Gaussian
452  // The Gaussian deviate is about 20% faster than Poisson, and for high N
453  // they are virtually identical.
454  const double MAX_POISSON=1.e5;
455  // Typedef for image row iterable
456  typedef typename ImageView<T>::iterator ImIter;
457 
458  data += T(_sky_level);
459 
460  // Add the Poisson noise first:
461  if (_gain > 0.) {
462  PoissonDeviate pd(*_rng, 1.); // will reset the mean for each pixel below.
463  GaussianDeviate gd(*_rng, 0., 1.);
464  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
465  ImIter ee = data.rowEnd(y);
466  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
467  if (*it <= 0.) continue;
468  double electrons = *it * _gain;
469  if (electrons < MAX_POISSON) {
470  pd.setMean(electrons);
471  *it = T(pd() / _gain);
472  } else {
473  gd.setSigma(sqrt(electrons)/_gain);
474  *it = T(*it + gd());
475  }
476  }
477  }
478  }
479

481  if (_read_noise > 0.) {
482  GaussianDeviate gd(*_rng, 0., _read_noise / (_gain > 0. ? _gain : 1.));
483  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
484  ImIter ee = data.rowEnd(y);
485  for (ImIter it = data.rowBegin(y); it != ee; ++it) {
486  *it = T(*it + gd());
487  }
488  }

505  void applyToVar(ImageView<T> data, ImageView<T> var)
506  {
507  // Typedef for image row iterable
508  typedef typename ImageView<T>::iterator ImIter;
509  assert(data.getBounds() == var.getBounds());
510  // Fill with the (constant) Gaussian contribution to variance
511  if (_read_noise > 0.) {
512  double sigma = _read_noise / (_gain > 0. ? _gain : 1.);
513  var.fill(sigma * sigma);
514  }
515  // Add the Poisson variance:
516  if (_gain > 0.) {
517  for (int y = data.getYMin(); y <= data.getYMax(); y++) { // iterate over y
518  ImIter ee = data.rowEnd(y);

520  for (ImIter it = data.rowBegin(y); it != ee; ++it, ++it2) {
521  if (*it > 0.) *it2 += (*it + _sky_level) / _gain;
522  }
523  }
524  }
