99  void buildTree(double threshold=0.)
100  {
101  dbg<<"buildTree\n";
102  assert(!empty());
103  assert(!_root);
104  // Sort the list so the largest flux regions are first.
105  std::sort(begin(), end(), FluxCompare());
106  VecIter start = begin();
107  VecIter last =
108  threshold == 0. ? end() :
109  std::upper_bound(begin(), end(), threshold, FluxCompare());
110  const int nelem = last-start;
111  dbg<<"N elements to build tree with = "<<nelem<<std::endl;
112  // Figure out what the total absolute flux is
113  // NB. Accumulate from end for better numerical accuracy adding up small values.
114  _totalAbsFlux = 0.;
115  for (VecIter it=last; it!=start;)
116  _totalAbsFlux += std::abs((--it)->getFlux());
117  dbg<<"totalAbsFlux = "<<_totalAbsFlux<<std::endl;
118  // leftAbsFlux will be updated for each element to be the total flux up the that one.
119  double leftAbsFlux = 0.;
120  _root = new Element(begin(), last, leftAbsFlux, _totalAbsFlux);
121  xdbg<<"Finished making root.\n";
122  xdbg<<"leftAbs = "<<leftAbsFlux<<", tot = "<<_totalAbsFlux<<std::endl;
123  xdbg<<"abs(diff) = "<<std::abs(leftAbsFlux - _totalAbsFlux)<<std::endl;
124  xdbg<<"cf. "<<1.e-8 * _totalAbsFlux<<std::endl;
125  xassert(std::abs(leftAbsFlux - _totalAbsFlux) <= 1.e-8 * _totalAbsFlux);
126  dbg<<"Done buildTree\n";
127 
128  // shortcut is a quick way to get to the right Element, or at least a better
129  // starting point, rather than always starting at root.
130  // We build this as we build the tree in the Element constructors.
131  _shortcut.resize(nelem,0);
132  buildShortcut(_root, 0, nelem);
133 #ifdef DEBUGLOGGING
134  // Make sure all the shortcut entries were set.
135  for(int i=0;i<nelem;++i) xassert(_shortcut[i]);
136 #endif
137  }

142  class Element
143  {
144  public:
145  Element(VecIter start, VecIter end, double& leftAbsFlux, double absFlux) :
146  _dataPtr(0), _left(0), _right(0),
147  _leftAbsFlux(leftAbsFlux), _absFlux(absFlux), _invAbsFlux(1./absFlux)
148  {
149  xassert(start != end);
150  if (start + 1 == end) {
151  // Only one element.
152  _dataPtr = &(*start);
153  // absFlux on input should equal the absolute flux in this dataPtr.
154  xassert(std::abs(std::abs(_dataPtr->getFlux()) - absFlux) <=
155  1.e-8 * (leftAbsFlux+absFlux));
156  // Update the running total of leftAbsFlux.
157  leftAbsFlux += std::abs(_dataPtr->getFlux());
158  } else if (start + 2 == end) {
159  // Two elements, so just split
160  VecIter mid = start+1;
161  _left = new Element(start, mid, leftAbsFlux, std::abs(start->getFlux()));
162  _right = new Element(mid, end, leftAbsFlux, std::abs(mid->getFlux()));
163  } else {
164  xassert(end > start+2);
165  xassert(absFlux > 0.);
166  VecIter mid = start;
167  // Divide the range by probability, not by number.
168  // The tree is intentionally unbalanced, so most of the time, the search
169  // stops quickly with the large flux Elements on the left.
170  double half_tot = absFlux/2.;
171  double leftSum=0.;
172  for (; leftSum < half_tot; ++mid) leftSum += std::abs(mid->getFlux());
173 
174  if (mid == end) {
175  dbg<<"mid passed the end. Backtracking...\n";
176  dbg<<"leftSum = "<<leftSum<<std::endl;
177  // Shouldn't happen in exact arithmetic, but just in case...
178  --mid;
179  leftSum -= std::abs(mid->getFlux());
180  dbg<<"leftSum => "<<leftSum<<std::endl;
181  }
182 
183  double rightSum;
184  if (leftSum > 0.9 * absFlux) {
185  // Then we're likely to start accumulating inaccuracies in absFlux
186  // if we just subtract, so recalculate. Indeed, numerical inaccuracies
187  // can make leftSum come out > absFlux, in which case the recalculation
188  // is definitely necessary.
189  dbg<<"leftSum = "<<leftSum<<std::endl;
190  dbg<<"absFlux = "<<absFlux<<std::endl;
191  dbg<<"leftAbsFlux = "<<leftAbsFlux<<std::endl;
192  dbg<<"absFlux - leftSum = "<<absFlux - leftSum<<std::endl;
193  rightSum = 0.;
194  for (VecIter it=end; it!=mid;) rightSum += std::abs((--it)->getFlux());
195  dbg<<"rightSum = "<<rightSum<<std::endl;
196  _absFlux = leftSum + rightSum;
197  dbg<<"leftSum + rightSum = "<<_absFlux<<std::endl;
198  } else {
199  // Otherwise, probably ok to just subtract
200  rightSum = absFlux - leftSum;
201  }
202 
203  _left = new Element(start, mid, leftAbsFlux, leftSum);
204  _right = new Element(mid, end, leftAbsFlux, rightSum);
205  // this element's absFlux should equal the sum of the two children's fluxes.
206  xassert(std::abs((_left->_absFlux + _right->_absFlux) - _absFlux) <
207  1.e-8 * (leftAbsFlux+_absFlux));
208  }
209  }

228  const FluxData* find(double& cumulativeFlux) const
229  {
230  xassert(cumulativeFlux >= _leftAbsFlux);
231  xassert(cumulativeFlux <= _leftAbsFlux + _absFlux);
232  if (!_left) {
233  xassert(!_right);
234  // This is a leaf. Answer is this element.
235  cumulativeFlux = (cumulativeFlux - _leftAbsFlux) * _invAbsFlux;
236  xassert(cumulativeFlux >= 0.);
237  xassert(cumulativeFlux <= 1.);
238  return _dataPtr;
239  } else {
240  xassert(_right);
241  if (cumulativeFlux < _right->_leftAbsFlux)
242  return _left->find(cumulativeFlux);
243  else
244  return _right->find(cumulativeFlux);
245  }
246  }

284  void buildShortcut(const Element* element, int i1, int i2)
285  {
286  // If i1 == i2, then we've already assigned everything, so stop recursing.
287  if (i1 == i2) return;
288 
289  // Figure out which bins in the shortcut vector should point to this element.
290  // On input, we are tasked with assigning indices i1 <= i < i2 to be either
291  // this element or one of its decendents.
292  xassert(i1*_totalAbsFlux/_shortcut.size() >= element->getLeftAbsFlux()-1.e-8);
293  xassert(i2*_totalAbsFlux/_shortcut.size() <=
294  element->getLeftAbsFlux()+element->getAbsFlux()+1.e-8);
295 
296  // If this is a node, then the only one we should assign is the shortcut
297  // bin that include both left and right. In other words the bin corresponding
298  // to the dividing flux.
299  if (element->isNode()) {
300  double f = element->getRight()->getLeftAbsFlux();
301  int imid = int(f * _shortcut.size() / _totalAbsFlux);
302  if (imid < i1) {
303  // Then the appropriate range is all in the right subtree
304  buildShortcut(element->getRight(), i1, i2);
305  } else if (imid >= i2) {
306  // Then the appropriate range is all in the left subtree
307  buildShortcut(element->getLeft(), i1, i2);
308  } else {
309  // Then there is an unassigned bin that spans both children.
310  // Set it to this element.
311  assert(imid >= i1);
312  assert(imid < i2);
313  _shortcut[imid] = element;
314  // Continue on with the sub-ranges on each side.
315  buildShortcut(element->getLeft(), i1, imid);
316  buildShortcut(element->getRight(), imid+1, i2);
317  }
318  } else {
319  // If we are at a leaf, then this leaf encompasses all the bins in the range.
320  // Assigne them all to this element.
321  for(int i=i1; i<i2; ++i) _shortcut[i] = element;
322  }
323  }
