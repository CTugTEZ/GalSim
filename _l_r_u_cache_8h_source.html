150  template <typename Key, typename Value>
151  class LRUCache
152  {
153  public:
159  LRUCache(size_t nmax) : _nmax(nmax) {}
160 
166  ~LRUCache() {}
167 
168  shared_ptr<Value> get(const Key& key)
169  {
170  assert(_entries.size() == _cache.size());
171  MapIter iter = _cache.find(key);
172  if (iter != _cache.end()) {
173  // Item is cached.
174  // Move it to the front of the list.
175  if (iter != _cache.begin())
176  _entries.splice(_entries.begin(), _entries, iter->second);
177  // Return the item's value
178  assert(_entries.size() == _cache.size());
179  return iter->second->second;
180  } else {
181  // Item is not cached.
182  // Make a new one.
183  shared_ptr<Value> value(LRUCacheHelper<Value,Key>::NewValue(key));
184  // Remove items from the cache as necessary.
185  while (_entries.size() >= _nmax) {
186  _cache.erase(_entries.back().first);
187  _entries.pop_back();
188  }
189  // Add the new value to the front.
190  _entries.push_front(Entry(key,value));
191  // Also put it in the cache
192  _cache[key] = _entries.begin();
193  // Return the new value
194  assert(_entries.size() == _cache.size());
195  return value;
196  }
197  }
198 
199  private:
200 
201  size_t _nmax;
202 
203  typedef std::pair<Key, shared_ptr<Value> > Entry;
204  std::list<Entry> _entries;
205 
206  typedef typename std::list<Entry>::iterator ListIter;
207  std::map<Key, ListIter> _cache;
208 
209  typedef typename std::map<Key, ListIter>::iterator MapIter;
210  };
