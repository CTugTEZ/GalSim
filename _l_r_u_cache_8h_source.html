122  boost::shared_ptr<Value> get(const Key& key)
123  {
124  assert(_entries.size() == _cache.size());
125  MapIter iter = _cache.find(key);
126  if (iter != _cache.end()) {
127  // Item is cached.
128  // Move it to the front of the list.
129  if (iter != _cache.begin())
130  _entries.splice(_entries.begin(), _entries, iter->second);
131  // Return the item's value
132  assert(_entries.size() == _cache.size());
133  return iter->second->second;
134  } else {
135  // Item is not cached.
136  // Make a new one.
137  boost::shared_ptr<Value> value(LRUCacheHelper<Value,Key>::NewValue(key));
138  // Remove items from the cache as necessary.
139  while (_entries.size() >= _nmax) {
140  bool erased = _cache.erase(_entries.back().first);
141  assert(erased);
142  _entries.pop_back();
143  }
144  // Add the new value to the front.
145  _entries.push_front(Entry(key,value));
146  // Also put it in the cache
147  _cache[key] = _entries.begin();
148  // Return the new value
149  assert(_entries.size() == _cache.size());
150  return value;
151  }
152  }
153
