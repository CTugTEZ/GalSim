549  template <class U>
550  void copyFrom(const BaseImage<U>& rhs) const
551  {
552  if (!this->_bounds.isSameShapeAs(rhs.getBounds()))
553  throw ImageError("Attempt im1 = im2, but bounds not the same shape");
554  for (int y=this->getYMin(), y2=rhs.getYMin(); y <= this->getYMax(); ++y, ++y2) {
555  iterator it1 = rowBegin(y);
556  const iterator ee = rowEnd(y);
557  typename BaseImage<U>::const_iterator it2 = rhs.rowBegin(y2);
558  while (it1 != ee) *(it1++) = T(*(it2++));
559  }
560  }
