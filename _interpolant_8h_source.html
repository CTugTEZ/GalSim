179  // Class that draws photons from this Interpolant
180  mutable shared_ptr<OneDimensionalDeviate> _sampler;

182  // Allocate photon sampler and do all of its pre-calculations
183  virtual void checkSampler() const
184  {
185  if (_sampler.get()) return;
186  // Will assume by default that the Interpolant kernel changes sign at non-zero
187  // integers, with one extremum in each integer range.
188  int nKnots = int(ceil(xrange()));
189  std::vector<double> ranges(2*nKnots);
190  for (int i=1; i<=nKnots; i++) {
191  double knot = std::min(double(i), xrange());
192  ranges[nKnots-i] = -knot;
193  ranges[nKnots+i-1] = knot;
194  }
195  _sampler.reset(new OneDimensionalDeviate(_interp, ranges, false, _gsparams));
196  }
