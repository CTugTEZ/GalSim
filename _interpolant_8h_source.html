188  // Allocate photon sampler and do all of its pre-calculations
189  virtual void checkSampler() const
190  {
191  if (_sampler.get()) return;
192  // Will assume by default that the Interpolant kernel changes sign at non-zero
193  // integers, with one extremum in each integer range.
194  int nKnots = int(ceil(xrange()));
195  std::vector<double> ranges(2*nKnots);
196  for (int i=1; i<=nKnots; i++) {
197  double knot = std::min(double(i), xrange());
198  ranges[nKnots-i] = -knot;
199  ranges[nKnots+i-1] = knot;
200  }
201  _sampler.reset(new OneDimensionalDeviate(_interp, ranges, false, _gsparams));
202  }
