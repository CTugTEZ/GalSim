350  struct CalculateCenter
351  {
352  static Position<U> call(const Bounds<T>& b)
353  { return Position<U>((b.getXMin() + b.getXMax())/U(2),(b.getYMin() + b.getYMax())/U(2)); }

355  // Slightly different for integer types:
356  template <class T, class U>
357  struct CalculateCenter<T, U, true>
358  {
359  static Position<U> call(const Bounds<T>& b)
360  {
361  // Write it this way to make sure the integer rounding goes the same way regardless
362  // of whether the values are positive or negative.
363  // e.g. (1,10,1,10) -> (6,6)
364  // (-10,-1,-10,-1) -> (-5,-5)
365  // Just up and to the right of the true center in both cases.
366  return Position<U>(b.getXMin() + (b.getXMax()-b.getXMin()+1)/U(2),
367  b.getYMin() + (b.getYMax()-b.getYMin()+1)/U(2));
368  }
369  };

397  template <class T>
398  std::vector< Bounds<T> > Bounds<T>::divide(int nx, int ny) const
399  {
400  if (!defined) return std::vector< Bounds<T> >(nx,ny);
401  typename std::vector< Bounds<T> > temp(nx*ny);
402  std::vector<double> x(nx+1);
403  std::vector<double> y(ny+1);
404  x[0] = xmin; x[nx] = xmax;
405  y[0] = ymin; y[ny] = ymax;
406  double xstep = (xmax-xmin)/nx;
407  double ystep = (ymax-ymin)/ny;
408  for(int i=1;i<nx;i++) x[i] = x[0]+i*xstep;
409  for(int j=1;j<ny;j++) y[j] = y[0]+j*ystep;
410  typename std::vector< Bounds<T> >::iterator ii=temp.begin();
411  for(int i=0;i<nx;i++) for(int j=0;j<ny;j++, ++i)
412  *ii = Bounds<T>(x[i],x[i+1],y[j],y[j+1]);
413  return temp;
414  }
