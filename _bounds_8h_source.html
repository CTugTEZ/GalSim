368  template <class T, class U, bool is_int>
369  struct CalculateCenter
370  {
371  static Position<U> call(const Bounds<T>& b)
372  { return Position<U>((b.getXMin() + b.getXMax())/U(2),(b.getYMin() + b.getYMax())/U(2)); }
373  };
374  // Slightly different for integer types:
375  template <class T, class U>
376  struct CalculateCenter<T, U, true>
377  {
378  static Position<U> call(const Bounds<T>& b)
379  {
380  // Write it this way to make sure the integer rounding goes the same way regardless
381  // of whether the values are positive or negative.
382  // e.g. (1,10,1,10) -> (6,6)
383  // (-10,-1,-10,-1) -> (-5,-5)
384  // Just up and to the right of the true center in both cases.
385  return Position<U>(b.getXMin() + (b.getXMax()-b.getXMin()+1)/U(2),
386  b.getYMin() + (b.getYMax()-b.getYMin()+1)/U(2));
387  }
388  };

416  template <class T>
417  std::vector< Bounds<T> > Bounds<T>::divide(int nx, int ny) const
418  {
419  if (!defined) return std::vector< Bounds<T> >(nx,ny);
420  typename std::vector< Bounds<T> > temp(nx*ny);
421  std::vector<double> x(nx+1);
422  std::vector<double> y(ny+1);
423  x[0] = xmin; x[nx] = xmax;
424  y[0] = ymin; y[ny] = ymax;
425  double xstep = (xmax-xmin)/nx;
426  double ystep = (ymax-ymin)/ny;
427  for(int i=1;i<nx;i++) x[i] = x[0]+i*xstep;
428  for(int j=1;j<ny;j++) y[j] = y[0]+j*ystep;
429  typename std::vector< Bounds<T> >::iterator ii=temp.begin();
430  for(int i=0;i<nx;i++) for(int j=0;j<ny;j++, ++i)
431  *ii = Bounds<T>(x[i],x[i+1],y[j],y[j+1]);
432  return temp;
433  }
