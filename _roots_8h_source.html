81  inline double getBesselRoot0(int s)
82  {
83  if (s <= 0)
84  throw std::runtime_error("s must be > 0");
85  if (s <= n_roots_j0) return root_j0[s-1];
86  else {
87  // Above this value, the asymptotic formula from Abramowitz and Stegun 9.5.12
88  // is accurate to better than 1.e-16, so essentially exact.
89  // b - (m-1)/8b - 4(m-1)(7m-31)/3(8b)^3 - 32(m-1)(83m^2-982m+3779)/15(8b)^5
90  // - 64(m-1)(6949m^3-153855m^2+1585743m-6277237)/105(8b)^7
91  // where m = 4nu^2 and b = (s + nu/2 - 1/4)pi.
92  //
93  // For nu = 0, this simplifies considerably to:
94  // b + 1/8b - 4*31/3(8b)^3 + 32*3779/15(8b)^5 - 64*6277237/105(8b)^7
95  double b = (s - 0.25) * M_PI;
96  double temp = 0.125/b;
97  double inv8bsq = temp*temp;
98 
99  // From here on, b will be the running total, not the original b, and temp will be
100  // the part of each coefficient that is in common with the next term.
101  b += temp;
102 
103  temp *= (4./3.) * inv8bsq;
104  b -= 31. * temp;
105 
106  temp *= (8./5.) * inv8bsq;
107  b += 3779. * temp;
108 
109  temp *= (2./7.) * inv8bsq;
110  b -= 6277237. * temp;
111 
112  return b;
113  }
114  }
