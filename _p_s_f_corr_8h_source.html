69  struct HSMParams {
70  /*
71  * @brief Parameters that determine how the moments/shape estimation routines make
72  * speed/accuracy tradeoff decisions.
73  *
74  * @param nsig_rg A parameter used to optimize convolutions by cutting off the galaxy
75  * profile. In the first step of the re-Gaussianization method of PSF
76  * correction, a Gaussian approximation to the pre-seeing galaxy is
77  * calculated. If re-Gaussianization is called with the flag 0x4 (as
78  * is the default), then this approximation is cut off at nsig_rg
79  * sigma to save computation time in convolutions.
80  * @param nsig_rg2 A parameter used to optimize convolutions by cutting off the PSF
81  * residual profile. In the re-Gaussianization method of PSF
82  * correction, a "PSF residual" (the difference between the true PSF
83  * and its best-fit Gaussian approximation) is constructed. If
84  * re-Gaussianization is called with the flag 0x8 (as is the default),
85  * then this PSF residual is cut off at nsig_rg2 sigma to save
86  * computation time in convolutions.
87  * @param max_moment_nsig2 A parameter for optimizing calculations of adaptive moments by
88  * cutting off profiles. This parameter is used to decide how many
89  * sigma^2 into the Gaussian adaptive moment to extend the moment
90  * calculation, with the weight being defined as 0 beyond this point.
91  * i.e., if max_moment_nsig2 is set to 25, then the Gaussian is
92  * extended to (r^2/sigma^2)=25, with proper accounting for elliptical
93  * geometry. If this parameter is set to some very large number, then
94  * the weight is never set to zero and the exponential function is
95  * always called. Note: GalSim script devel/modules/test_mom_timing.py
96  * was used to choose a value of 25 as being optimal, in that for the
97  * cases that were tested, the speedups were typically factors of
98  * several, but the results of moments and shear estimation were
99  * changed by <10^-5. Not all possible cases were checked, and so for
100  * use of this code for unusual cases, we recommend that users check
101  * that this value does not affect accuracy, and/or set it to some
102  * large value to completely disable this optimization.
103  * @param regauss_too_small A parameter for how strictly the re-Gaussianization code treats
104  * small galaxies. If this parameter is 1, then the re-Gaussianization
105  * code does not impose a cut on the apparent resolution before trying
106  * to measure the PSF-corrected shape of the galaxy; if 0, then it is
107  * stricter. Using the default value of 1 prevents the
108  * re-Gaussianization PSF correction from completely failing at the
109  * beginning, before trying to do PSF correction, due to the crudest
110  * possible PSF correction (Gaussian approximation) suggesting that
111  * the galaxy is very small. This could happen for some usable
112  * galaxies particularly when they have very non-Gaussian surface
113  * brightness profiles -- for example, if there's a prominent bulge
114  * that the adaptive moments attempt to fit, ignoring a more
115  * extended disk. Setting a value of 1 is useful for keeping galaxies
116  * that would have failed for that reason. If they later turn out to
117  * be too small to really use, this will be reflected in the final
118  * estimate of the resolution factor, and they can be rejected after
119  * the fact.
120  * @param adapt_order The order to which circular adaptive moments should be calculated
121  * for KSB method. This parameter only affects calculations using the
122  * KSB method of PSF correction. Warning: deviating from default
123  * value of 2 results in code running more slowly, and results have
124  * not been significantly tested.
125  * @param convergence_threshold Accuracy (in x0, y0, and sigma as a fraction of sigma) when
126  * calculating adaptive moments.
127  * @param max_mom2_iter Maximum number of iterations to use when calculating adaptive
128  * moments. This should be sufficient in nearly all situations, with
129  * the possible exception being very flattened profiles.
130  * @param num_iter_default Number of iterations to report in the output ShapeData structure
131  * when code fails to converge within max_mom2_iter iterations.
132  * @param bound_correct_wt Maximum shift in centroids and sigma between iterations for
133  * adaptive moments.
134  * @param max_amoment Maximum value for adaptive second moments before throwing
135  * exception. Very large objects might require this value to be
136  * increased.
137  * @param max_ashift Maximum allowed x / y centroid shift (units: pixels) between
138  * successive iterations for adaptive moments before throwing
139  * exception.
140  * @param ksb_moments_max Use moments up to ksb_moments_max order for KSB method of PSF
141  * correction.
142  * @param ksb_sig_weight The width of the weight function (in pixels) to use for the KSB
143  * method. Normally, this is derived from the measured moments of the
144  * galaxy image; this keyword overrides this calculation. Can be
145  * combined with ksb_sig_factor.
146  * @param ksb_sig_factor Factor by which to multiply the weight function width for the KSB
147  * method (default: 1.0). Can be combined with ksb_sig_weight.
148  * @param failed_moments Value to report for ellipticities and resolution factor if shape
149  * measurement fails.
150  */
151  HSMParams(double _nsig_rg,
152  double _nsig_rg2,
153  double _max_moment_nsig2,

163  double _ksb_sig_weight,
